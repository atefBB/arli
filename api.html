# arli.transform( value, [options] ) إلى أعلى

  • Returns: String

تمكنك هذه الميزة من تحويل النصوص الشبه عربية إلى عربية. 


arli.transform('مرحبا بالعـــــــــــــالم');
// -> 'مرحبا بالعـالم'

# value إلى أعلى

  • Type: String
  • Default: Undefined

يستعمل لإدراج القيمة أو النص المراد تحويله. 


arli.transform('مرحبا بالعـــــــــــــالم');
// -> 'مرحبا بالعـالم'

# [options] إلى أعلى

  • Type: Object
  • Default: {clean: true, date: true, dateFrom: 'all', dateTo: 'all', digit: true, excludeLigatures: [' '], excludePunc: ' ', ligatures: true, numberSep: true, punc: true, removeTatweel: 'extra'}

يستعمل كخيارات لتحويل النص، سيتم شرحها تفصيلا فيما يلي. 


arli.transform('0123456789', {digit: false});
// -> '0123456789'

# [options.clean] إلى أعلى

  • Type: Boolean
  • Default: true

ينزع الشوائب الزائدة عن النص مثل المساحات المبالغ فيها. 


arli.transform('     مرحبا    ', {clean: true});
// -> 'مرحبا'

arli.transform('     مرحبا    ', {clean: false});
// -> '     مرحبا     '

# [options.date] إلى أعلى

  • Type: Boolean
  • Default: true

إتاحة أو تعطيل تحويل التاريخ إلى الصيغة العربية. 


arli.transform('27/05/1989 27.05.1989 27-05-1989', {date: true});
// -> '٢٧؍٠٥؍١٩٨٩ ٢٧؍٠٥؍١٩٨٩ ٢٧؍٠٥؍١٩٨٩'

arli.transform('27/05/1989 27.05.1989 27-05-1989', {date: false});
// -> '27/05/1989 27.05.1989 27-05-1989'

# [options.dateFrom] إلى أعلى

  • Type: String
  • Default: 'all'
  • Options: 'all', 'DMY', 'MDY'

إختيار صيغة التاريخ المراد تحويلها. 


arli.transform('27/05/1989 05/27/1989', {dateFrom: 'all'});
// -> '٢٧؍٠٥؍١٩٨٩ ٠٥؍٢٧؍١٩٨٩'

arli.transform('27/05/1989 05/27/1989', {dateFrom: 'DMY'});
// -> '٢٧؍٠٥؍١٩٨٩ 05/27/1989'

arli.transform('27/05/1989 05/27/1989', {dateFrom: 'MDY'});
// -> '27/05/1989 ٠٥؍٢٧؍١٩٨٩'

# [options.dateTo] إلى أعلى

  • Type: String
  • Default: 'all'
  • Options: 'all', 'DMY', 'MDY'

إختيار صيغة التاريخ المراد التحويل إليها. 


arli.transform('27/05/1989 05/27/1989', {dateTo: 'all'});
// -> '٢٧؍٠٥؍١٩٨٩ ٠٥؍٢٧؍١٩٨٩'

arli.transform('27/05/1989 05/27/1989', {dateTo: 'DMY'});
// -> '٢٧؍٠٥؍١٩٨٩ ٢٧؍٠٥؍١٩٨٩'

arli.transform('27/05/1989 05/27/1989', {dateTo: 'MDY'});
// -> '٠٥؍٢٧؍١٩٨٩ ٠٥؍٢٧؍١٩٨٩'

# [options.digit] إلى أعلى

  • Type: Boolean
  • Default: true

تحويل الأرقام إلى عربية هندية. 


arli.transform('0123456789', {digit: true});
// -> '٠١٢٣٤٥٦٧٨٩'

arli.transform('0123456789', {digit: false});
// -> '0123456789'

# [options.excludeLigatures] إلى أعلى

  • Type: Array
  • Default: [' ']

إقصاء بعض الجمل الطويلة من التقليص.
المثال أدناه يرى من اليمين إلى اليسار: محمد ﷺ و عمر ؓ 


arli.transform('محمد صلى الله عليه و سلم و عمر رضي الله عنه', {excludeLigatures: [' ']});
// -> 'محمد ﷺ و عمر ؓ'

arli.transform('محمد صلى الله عليه و سلم و عمر رضي الله عنه', {excludeLigatures: ['عمر رضي الله عنه']});
// -> 'محمد ﷺ و عمر رضي الله عنه'

# [options.excludePunc] إلى أعلى

  • Type: String
  • Default: ' '

إقصاء بعض الرموز من التحويل. 


arli.transform(')?,;%(', {excludePunc: ' '});
// -> '﴿٪؟،؛﴾'

arli.transform(')?,;%(', {excludePunc: '()%'});
// -> ')%؟،؛('

# [options.ligatures] إلى أعلى

  • Type: Boolean
  • Default: true

تقليص الجمل الطويلة. 


arli.transform('محمد صلى الله عليه و سلم', {ligatures: true});
// -> 'محمد ﷺ'

arli.transform('محمد صلى الله عليه و سلم', {ligatures: false});
// -> 'محمد صلى الله عليه و سلم'

# [options.numberSep] إلى أعلى

  • Type: Boolean
  • Default: true

تحويل فواصل الأرقام إلى الصيغة العربية. 


arli.transform('10.50 999,888.00', {numberSep: true});
// -> '١٠٫٥٠ ٩٩٩٬٨٨٨٫٠٠'

arli.transform('10.50 999,888.00', {numberSep: false});
// -> '١٠.٥٠ ٩٩٩،٨٨٨.٠٠'

# [options.punc] إلى أعلى

  • Type: Boolean
  • Default: true

تحويل الرموز إلى الصيغ العربية. 


arli.transform(')?,;%(', {punc: true});
// -> '﴿٪؟،؛﴾'

arli.transform(')?,;%(', {punc: false});
// -> ')?,;%('

# [options.removeTatweel] إلى أعلى

  • Type: String
  • Default: 'extra'
  • Options: 'extra', 'all', 'none'

مسح المد المبالغ فيه. 


arli.transform('مرحبا بالعـــــــالم', {removeTatweel: 'extra'});
// -> 'مرحبا بالعـالم'

arli.transform('مرحبا بالعـــــــالم', {removeTatweel: 'all'});
// -> 'مرحبا بالعالم'

arli.transform('مرحبا بالعـــــــالم', {removeTatweel: 'none'});
// -> 'مرحبا بالعـــــــالم'
